#########
# 3D PDF Method 2: OSG + libPRC + libHaru (optional)
#########
option(EXPORT_3DPDF_USE_LIBPRC "Enable 3D PDF export via OSG + libPRC + libHaru (Method 2)" OFF)

if(EXPORT_3DPDF_USE_LIBPRC)
    # --- Library root directories (user configures these) ---
    set(DEFAULT_LIBPRC_DIR  "${CMAKE_SOURCE_DIR}/3rdparty/libPRC")
    set(DEFAULT_LIBHARU_DIR "${CMAKE_SOURCE_DIR}/3rdparty/libharu-2.3.0")
    set(OSG_DIR        CACHE PATH "Path to OpenSceneGraph root (for osgdb_prc plugin only)")
    set(LIBPRC_DIR     "${DEFAULT_LIBPRC_DIR}"  CACHE PATH "Path to libPRC root")
    set(LIBHARU_DIR    "${DEFAULT_LIBHARU_DIR}" CACHE PATH "Path to libharu root")

    # --- Includes ---
    set(LIBPRC_INCLUDE_DIRS
        "${LIBPRC_DIR}/include"
        "${LIBPRC_DIR}/include/asymptote"
        "${LIBPRC_DIR}/include/libPRC"
        "${LIBHARU_DIR}/include"
    )

    # --- Libraries ---
    find_library(PRC_LIB_RELEASE      NAMES libPRC      PATHS "${LIBPRC_DIR}/lib" NO_DEFAULT_PATH)
    find_library(PRC_LIB_DEBUG        NAMES libPRCd     PATHS "${LIBPRC_DIR}/lib" NO_DEFAULT_PATH)
    find_library(ASYMPTOTE_LIB_RELEASE NAMES asymptote  PATHS "${LIBPRC_DIR}/lib" NO_DEFAULT_PATH)
    find_library(ASYMPTOTE_LIB_DEBUG   NAMES asymptoted PATHS "${LIBPRC_DIR}/lib" NO_DEFAULT_PATH)

    # libharu: used to embed the PRC file into a PDF 
    find_library(HPDF_LIB  NAMES libhpdf  PATHS "${LIBHARU_DIR}/lib" NO_DEFAULT_PATH)
    if(NOT HPDF_LIB)
        message(WARNING "libharu (libhpdf.lib) not found at ${LIBHARU_DIR}/lib. PDF embedding will fail.")
    endif()
    message(STATUS "  libharu: ${HPDF_LIB}")

    # asymptote depends on zlib (deflate* symbols). Use the one from OCCT 3rdparty.
    find_library(ZLIB_LIB
        NAMES zlib
        PATHS "${OpenCASCADE_3RDPARTY_DIR}/zlib-1.2.8-vc14-64/lib"
        NO_DEFAULT_PATH
    )
    if(NOT ZLIB_LIB)
        # fallback: search common locations
        find_library(ZLIB_LIB NAMES zlib zdll)
    endif()
    if(NOT ZLIB_LIB)
        message(WARNING "zlib not found; asymptote.lib may have unresolved deflate* symbols.")
    endif()

    # Require at least the Release variants
    if(NOT PRC_LIB_RELEASE OR NOT ASYMPTOTE_LIB_RELEASE)
        message(FATAL_ERROR
            "EXPORT_3DPDF_USE_LIBPRC is ON but one or more required libraries were not found.\n"
            "  PRC_LIB_RELEASE:       ${PRC_LIB_RELEASE}\n"
            "  ASYMPTOTE_LIB_RELEASE: ${ASYMPTOTE_LIB_RELEASE}\n"
            "Please set LIBPRC_DIR correctly, or turn off EXPORT_3DPDF_USE_LIBPRC."
        )
    endif()

    # All libraries verified — mark the feature as enabled.
    set(LIBPRC_ENABLED TRUE)
    message(STATUS "[Method2-PRC] Libraries found. USE_LIBPRC will be defined.")
    message(STATUS "  PRC Release:       ${PRC_LIB_RELEASE}")
    message(STATUS "  PRC Debug:         ${PRC_LIB_DEBUG}")
    message(STATUS "  asymptote Release: ${ASYMPTOTE_LIB_RELEASE}")
    message(STATUS "  asymptote Debug:   ${ASYMPTOTE_LIB_DEBUG}")
    message(STATUS "  zlib:              ${ZLIB_LIB}")

    # Use optimized/debug keywords so the linker picks the correct CRT variant.
    # If a debug variant is missing, fall back to the release variant.
    if(NOT PRC_LIB_DEBUG)
        set(PRC_LIB_DEBUG ${PRC_LIB_RELEASE})
    endif()
    if(NOT ASYMPTOTE_LIB_DEBUG)
        set(ASYMPTOTE_LIB_DEBUG ${ASYMPTOTE_LIB_RELEASE})
    endif()

    set(LIBPRC_LIBRARIES
        optimized ${PRC_LIB_RELEASE}       debug ${PRC_LIB_DEBUG}
        optimized ${ASYMPTOTE_LIB_RELEASE} debug ${ASYMPTOTE_LIB_DEBUG}
        ${HPDF_LIB}
        ${ZLIB_LIB}
    )
endif()
